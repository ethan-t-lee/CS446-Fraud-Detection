1) Set Variables:
export PROJECT_ID="cs446-fraud-detection"
export REGION="us-central1"
export SERVICE="fraud-processor"
export REPO="fraud-repo"
export IMAGE="$REGION-docker.pkg.dev/$PROJECT_ID/$REPO/$SERVICE"

gcloud config set project "$PROJECT_ID"

2) Rebuild the container:
gcloud builds submit service --tag "$IMAGE"

3) Redeploy Cloud Run:
gcloud run deploy "$SERVICE" \
  --image "$IMAGE" \
  --region "$REGION" \
  --no-allow-unauthenticated
